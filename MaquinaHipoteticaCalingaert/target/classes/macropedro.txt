START   NOP
        MACRO
        INIT    &REG, &VAL
        LOAD    &VAL
        STORE   &REG    
        MEND
        MACRO
        SOMA    &REG1, &REG2, &RES
        LOAD    &REG1
        ADD     &REG2
        STORE   &RES
        MEND
        INIT    R1, 10
        INIT    R2, 20
        SOMA    R1, R2, R3
HALT    STOP
R1	SPACE
R2	SPACE
R3	SPACE
        END
